ANSI C Data Structures Library by Iuri Sônego Cardoso
Exemplo de uso de Fila. 

#define CAPACITY 5
//// create the queue.
QUEUE q = queueCreate(CAPACITY, sizeof(int));
int e;
//// adding 10 in the queue.
e = 10;
if (! queueAdd(q, &e)) {
     //// error: queue is full.
}
//// now, adding 20 and 30 in the queue.
e = 20;
queueAdd(q, &e);
e = 30;
queueAdd(q, &e);
//// here, the queue has 10, 20, 30.
//// print the size: 3.
printf("Queue size: %d\n", queueSize(q));
//// print: 10 20 30
while (queueRemove(q, &e)) {
   printf("%d ", e);
}
//// don't forget this, to avoid memory leak.
queueFree(q);
